% Faithfulness note: why the formal object is the intended one.
\newmdenv[backgroundcolor=faithbg, linecolor=faithframe,
          linewidth=0.6pt, innertopmargin=4pt, innerbottommargin=4pt,
          skipabove=4pt]{faithbox}
\newenvironment{faithnote}
  {\begin{faithbox}\noindent\textbf{Why this is the right definition.}\ }
  {\end{faithbox}}

% Closure panel: everything this statement needs.
\newmdenv[backgroundcolor=panelbg, linecolor=panelframe,
          linewidth=0.6pt, innertopmargin=4pt, innerbottommargin=4pt,
          skipabove=4pt]{panelbox}
\newenvironment{closurepanel}
  {\begin{panelbox}\noindent\textbf{Everything this statement needs}%
   \footnote{This list is \emph{generated} from the build: a script
   computes every constant the formal statement references,
   transitively through library definitions, and CI fails if any of
   them lacks an entry in the Dictionary
   (\texttt{scripts/statement\_closure.py}).}%
   \begin{itemize}\setlength\itemsep{0pt}}
  {\end{itemize}\end{panelbox}}
